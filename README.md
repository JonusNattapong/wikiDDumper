### 3. р╕кр╕│р╕лр╕гр╕▒р╕Ъ Text Mining р╣Бр╕ер╕░ Topic Modeling

```python
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.decomposition import LatentDirichletAllocation
import pandas as pd

# р╣Вр╕лр╕ер╕Фр╕Вр╣Йр╕нр╕бр╕╣р╕е
df = pd.read_csv("output/thwiki_articles_20250617.csv")

vectorizer = TfidfVectorizer(max_features=1000, stop_words=None)
doc_term_matrix = vectorizer.fit_transform(df['text'].head(1000))

# LDA Topic Modeling
lda = LatentDirichletAllocation(n_components=10, random_state=42)
lda.fit(doc_term_matrix)

# р╣Бр╕кр╕Фр╕Зр╕лр╕▒р╕зр╕Вр╣Йр╕н
feature_names = vectorizer.get_feature_names_out()
for topic_idx, topic in enumerate(lda.components_):
    top_words = [feature_names[i] for i in topic.argsort()[-10:]]
    print(f"р╕лр╕▒р╕зр╕Вр╣Йр╕н {topic_idx}: {', '.join(top_words)}")
```
